DirectoryClickedEventArgs

Just an EventArgs passing a DirectoryInfo.

using System;using System.IO;
 
namespace Library.Forms
{
    public class DirectoryClickedEventArgs : EventArgs
    {
        private DirectoryInfo _data;
        public DirectoryClickedEventArgs(DirectoryInfo data)
        {
            _data = data;
        }
 
        public DirectoryInfo Directory
        {
            get { return _data; }
            set { _data = value; }
        }
    }
}
